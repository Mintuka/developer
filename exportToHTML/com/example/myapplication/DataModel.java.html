17   package com.example.myapplication;
18   
19   class DataModel {
20   
21       // Member variables representing the title and information about the sport.
22       private String title;
23       private String info;
24       private int imageResource;
25   
26   
27       DataModel(String title, String info, int imageResource) {
28           this.title = title;
29           this.info = info;
30           this.imageResource = imageResource;
31       }
32   
33       public String getTitle() {
34           return title;
35       }
36   
37       public String getInfo() {
38           return info;
39       }
40   
41       public int getImageResource() {
42   
43           return imageResource;
44       }
45   }
