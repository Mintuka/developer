16   package com.example.myapplication;
17   
18   import android.content.Context;
19   import android.content.Intent;
20   import android.view.LayoutInflater;
21   import android.view.View;
22   import android.view.ViewGroup;
23   import android.widget.ImageView;
24   import android.widget.TextView;
25   import androidx.recyclerview.widget.RecyclerView;
26   import com.bumptech.glide.Glide;
27   import java.util.ArrayList;
28   
29   
30   class Adapter extends RecyclerView.Adapter<Adapter.ViewHolder>  {
31   
32       // Member variables.
33       private ArrayList<DataModel> LaptopsData;
34       private Context context;
35   
36       Adapter(Context context, ArrayList<DataModel> LaptopsData) {
37           this.LaptopsData = LaptopsData;
38           this.context = context;
39       }
40   
41   
42   
43       @Override
44       public Adapter.ViewHolder onCreateViewHolder(
45               ViewGroup parent, int viewType) {
46           return new ViewHolder(LayoutInflater.from(context).
47                   inflate(R.layout.list_item, parent, false));
48       }
49   
50   
51       @Override
52       public void onBindViewHolder(Adapter.ViewHolder holder,
53                                    int position) {
54           // Get current sport.
55           DataModel currentLaptop = LaptopsData.get(position);
56   
57           // Populate the textviews with data.
58           holder.bindTo(currentLaptop);
59       }
60   
61       @Override
62       public int getItemCount() {
63           return LaptopsData.size();
64       }
65   
66   
67   
68       class ViewHolder extends RecyclerView.ViewHolder implements View.OnClickListener{
69   
70           // Member Variables for the TextViews
71           private TextView TitleText;
72           private TextView InfoText;
73           private ImageView LaptopsImage;
74           @Override
75           public void onClick(View view) {
76               DataModel currentLaptop = LaptopsData.get(getAdapterPosition());
77               Intent detailIntent = new Intent(context, Detail.class);
78               detailIntent.putExtra("title", currentLaptop.getTitle());
79               detailIntent.putExtra("image_resource",
80                       currentLaptop.getImageResource());
81               context.startActivity(detailIntent);
82           }
83   
84           ViewHolder(View itemView) {
85               super(itemView);
86   
87               // Initialize the views.
88               TitleText = itemView.findViewById(R.id.title);
89               InfoText = itemView.findViewById(R.id.subTitle);
90               LaptopsImage = itemView.findViewById(R.id.LaptopsImage);
91   
92               itemView.setOnClickListener(this);
93           }
94   
95           void bindTo(DataModel currentLaptop){
96               // Populate the textviews with data.
97               TitleText.setText(currentLaptop.getTitle());
98               InfoText.setText(currentLaptop.getInfo());
99               Glide.with(context).load(currentLaptop.getImageResource()).into(LaptopsImage);
100          }
101      }
102  }
